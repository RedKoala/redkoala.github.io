/* A1 Colleen Carroll - Blinking LEDs */ // define the GPIO pins connected to each of the three colored LEDs const int GREEN_PIN = 2; const int YELLOW_PIN = 7; const int BLUE_PIN = 12; // define the delay in between each blink. const int DELAY = 500; // the setup function runs once when you press reset or power the board void setup() { // initialize the three pins connected to LEDs as output. pinMode(GREEN_PIN, OUTPUT); pinMode(YELLOW_PIN, OUTPUT); pinMode(BLUE_PIN, OUTPUT); } // alternate turning each of three colored LEDs on and off in the loop function void loop() { // turn the green LED on (HIGH is the voltage level) digitalWrite(GREEN_PIN, HIGH); // leave the green LED on for DELAY ms delay(DELAY); // turn the green LED off by making the voltage LOW digitalWrite(GREEN_PIN, LOW); // leave no delay in between turning one light off and the next on // turn the yellow LED on (HIGH is the voltage level) digitalWrite(YELLOW_PIN, HIGH); // leave the yellow LED on for DELAY ms delay(DELAY); // turn the yellow LED off by making the voltage LOW digitalWrite(YELLOW_PIN, LOW); // leave no delay in between turning one light off and the next on // turn the blue LED on (HIGH is the voltage level) digitalWrite(BLUE_PIN, HIGH); // leave the blue LED on for DELAY ms delay(DELAY); // turn the blue LED off by making the voltage LOW digitalWrite(BLUE_PIN, LOW); // leave no delay in between turning one light off and the next on }
